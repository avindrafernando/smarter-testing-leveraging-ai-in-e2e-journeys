Lessons Learned

😟 Overly relying on time based waits

it('should display a list of todos', () => {
  cy.visit('/')
  cy.wait(1000)
  cy.get('#my-todos').should('be.visible')
})

it('should load records', () => {
  cy.visit('/')
  cy.wait(5000)
  cy.get('#load-records').click()
  cy.wait(7500)
  cy.get('#my-records').should('be.visible')
})

✅ Spy using cy.intercept

it('should display a list of todos', () => {
  cy.intercept('GET', '/todos').as('todos')
  cy.visit('/')
  cy.wait('@todos')
  cy.get('#my-todos').should('be.visible')
})

✅ waitForURL to Navigate away

// Clicks a link
await page.click('a.some-link')

// Waits for the next page to load
await page.waitForURL('**/target.html')

✅ Wait for a network request/response

// Example: Waiting for a specific API response
await page.waitForResponse('https://example.com/api/data')

// Example: Waiting for any request to a certain endpoint
await page.waitForRequest('https://example.com/api/*')

 💡 Inadequate selectors

cy.get('button').click()              ❌

cy.get('.btn.btn-large').click()      ❌

cy.get('#main').click()               ⚠️

cy.get('[name="submission"]').click() ⚠️

cy.contains('Submit').click()         ✅

cy.get('[data-cy="submit"]').click()  ✅
